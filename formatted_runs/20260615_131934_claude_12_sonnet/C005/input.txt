package com.meridianbank.loanorigination.entity;

import jakarta.persistence.*;
import lombok.Getter;
import lombok.Setter;
import lombok.NoArgsConstructor;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

@Entity
@Table(name = "applicant")
@Getter
@Setter
@NoArgsConstructor
public class Applicant {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    @Column(name = "id")
    private Long id;

    @Column(name = "full_name", nullable = false, length = 150)
    private String fullName;

    @Column(name = "national_id_number", nullable = false, unique = true, length = 30)
    private String nationalIdNumber;

    @Column(name = "date_of_birth", nullable = false)
    private LocalDate dateOfBirth;

    @Column(name = "email", length = 150)
    private String email;

    @Column(name = "annual_income", precision = 15, scale = 2)
    private java.math.BigDecimal annualIncome;

    @Column(name = "employment_status", length = 30)
    private String employmentStatus;

    @Column(name = "registered_at", nullable = false)
    private LocalDateTime registeredAt;

    @OneToMany(mappedBy = "applicant", cascade = CascadeType.ALL, fetch = FetchType.LAZY)
    private List<LoanApplication> loanApplications = new ArrayList<>();

    @PrePersist
    protected void onCreate() {
        if (this.registeredAt == null) {
            this.registeredAt = LocalDateTime.now();
        }
    }

    public int calculateAge() {
        if (this.dateOfBirth == null) {
            return 0;
        }
        return java.time.Period.between(this.dateOfBirth, LocalDate.now()).getYears();
    }

    public boolean isEmployed() {
        return "EMPLOYED".equalsIgnoreCase(this.employmentStatus) || "SELF_EMPLOYED".equalsIgnoreCase(this.employmentStatus);
    }
}
