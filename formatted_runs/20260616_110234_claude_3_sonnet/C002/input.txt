package com.credex.loanorigination.dto;

import com.credex.loanorigination.entity.LoanType;
import jakarta.validation.constraints.DecimalMax;
import jakarta.validation.constraints.DecimalMin;
import jakarta.validation.constraints.Digits;
import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Pattern;
import jakarta.validation.constraints.Positive;
import jakarta.validation.constraints.Size;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;
import java.math.BigDecimal;

@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
public class LoanApplicationRequestDTO {

    @NotBlank(message = "Applicant ID must not be blank")
    @Pattern(regexp = "^APL-[0-9]{8}$", message = "Applicant ID must match pattern APL-XXXXXXXX")
    private String applicantId;

    @NotNull(message = "Requested amount is required")
    @Positive(message = "Requested amount must be a positive value")
    @DecimalMin(value = "10000.00", message = "Minimum loan amount is 10,000")
    @DecimalMax(value = "5000000.00", message = "Maximum loan amount is 5,000,000")
    @Digits(integer = 13, fraction = 2, message = "Requested amount must have at most 13 integer digits and 2 decimal places")
    private BigDecimal requestedAmount;

    @NotNull(message = "Tenure is required")
    @Min(value = 6, message = "Minimum tenure is 6 months")
    @Max(value = 360, message = "Maximum tenure is 360 months")
    private Integer tenureMonths;

    @NotNull(message = "Loan type is required")
    private LoanType loanType;

    @Size(max = 500, message = "Purpose must not exceed 500 characters")
    private String purpose;

    @NotBlank(message = "Branch code is required")
    @Size(min = 4, max = 10, message = "Branch code must be between 4 and 10 characters")
    private String branchCode;

    @NotBlank(message = "Relationship manager ID is required")
    @Pattern(regexp = "^RM-[0-9]{6}$", message = "Relationship manager ID must match pattern RM-XXXXXX")
    private String relationshipManagerId;

    @NotNull(message = "Self-declaration of no existing default is required")
    private Boolean selfDeclaredNoDefault;

    @DecimalMin(value = "0.0000", inclusive = true, message = "Declared monthly income must be zero or greater")
    @DecimalMax(value = "99999999.9999", message = "Declared monthly income exceeds permissible limit")
    @Digits(integer = 8, fraction = 4, message = "Declared monthly income must have at most 8 integer digits and 4 decimal places")
    private BigDecimal declaredMonthlyIncome;

    @DecimalMin(value = "0.0000", inclusive = true, message = "Declared monthly obligations must be zero or greater")
    @DecimalMax(value = "99999999.9999", message = "Declared monthly obligations exceed permissible limit")
    @Digits(integer = 8, fraction = 4, message = "Declared monthly obligations must have at most 8 integer digits and 4 decimal places")
    private BigDecimal declaredMonthlyObligations;
}
