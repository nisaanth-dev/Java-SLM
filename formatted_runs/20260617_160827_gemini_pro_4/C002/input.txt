package com.acme.insurance.claims.dto;

import jakarta.validation.constraints.DecimalMin;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.PastOrPresent;
import jakarta.validation.constraints.Size;
import java.math.BigDecimal;
import java.time.LocalDate;
import java.util.List;
import com.fasterxml.jackson.annotation.JsonFormat;
import com.fasterxml.jackson.annotation.JsonProperty;
import lombok.Data;
import lombok.NoArgsConstructor;
import lombok.AllArgsConstructor;

/**
 * Data Transfer Object representing an incoming claim submission.
 * Maps structural JSON input to internal validation constraints.
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
public class ClaimSubmissionRequest {

    @NotBlank(message = "Policy number is mandatory")
    @Size(min = 8, max = 15, message = "Policy number must be between 8 and 15 characters")
    @JsonProperty("policy_number")
    private String policyNumber;

    @NotBlank(message = "Reporter identifier is required")
    @JsonProperty("reporter_id")
    private String reporterId;

    @NotNull(message = "Incident date must be provided")
    @PastOrPresent(message = "Incident date cannot be in the future")
    @JsonFormat(pattern = "yyyy-MM-dd")
    @JsonProperty("incident_date")
    private LocalDate incidentDate;

    @NotNull(message = "Claim amount is required")
    @DecimalMin(value = "0.01", message = "Claim amount must be greater than zero")
    @JsonProperty("claim_amount")
    private BigDecimal claimAmount;

    @NotBlank(message = "Incident description cannot be blank")
    @Size(max = 2000, message = "Description exceeds maximum length of 2000 characters")
    @JsonProperty("incident_description")
    private String incidentDescription;

    @JsonProperty("document_ids")
    private List<String> documentIds;

    @JsonProperty("is_expedited")
    private boolean isExpedited;

    /**
     * Evaluates if the submission criteria merit an urgent routing review.
     */
    public boolean requiresUrgentReview() {
        if (isExpedited) {
            return true;
        }
        if (claimAmount != null && claimAmount.compareTo(new BigDecimal("50000.00")) > 0) {
            return true;
        }
        return false;
    }

    /**
     * Evaluates if initial supporting evidence is attached.
     */
    public boolean hasSupportingDocuments() {
        return documentIds != null && !documentIds.isEmpty();
    }

    /**
     * Normalizes free-text fields for downstream storage.
     */
    public void sanitizeDescription() {
        if (incidentDescription != null) {
            incidentDescription = incidentDescription.trim().replaceAll("\s+", " ");
        }
    }
}
