package com.acme.insurance.claims.exception;

import com.acme.insurance.claims.dto.ClaimResponse;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import java.util.stream.Collectors;

/**
 * Intercepts uncaught exceptions globally across all REST controllers.
 * Translates technical errors into standardized business responses.
 */
@RestControllerAdvice
public class GlobalExceptionHandler {

    private static final Logger log = LoggerFactory.getLogger(GlobalExceptionHandler.class);

    /**
     * Handles internal business domain exceptions explicitly thrown by the service layer.
     */
    @ExceptionHandler(ClaimSystemException.class)
    public ResponseEntity<ClaimResponse> handleClaimSystemException(ClaimSystemException ex) {
        log.error("Domain exception occurred: Code={}, Message={}", ex.getErrorCode(), ex.getMessage(), ex);

        ClaimResponse errorResponse = new ClaimResponse();
        errorResponse.setClaimStatus("SYSTEM_ERROR");

        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body(errorResponse);
    }

    /**
     * Handles framework-level payload validation failures (e.g., @Valid constraint violations).
     */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ClaimResponse> handleValidationException(MethodArgumentNotValidException ex) {
        String validationErrors = ex.getBindingResult().getFieldErrors().stream()
            .map(error -> error.getField() + ": " + error.getDefaultMessage())
            .collect(Collectors.joining(", "));

        log.warn("Payload validation failed: {}", validationErrors);

        ClaimResponse errorResponse = new ClaimResponse();
        errorResponse.setClaimStatus("INVALID_PAYLOAD");

        return ResponseEntity.status(HttpStatus.BAD_REQUEST).body(errorResponse);
    }
}
