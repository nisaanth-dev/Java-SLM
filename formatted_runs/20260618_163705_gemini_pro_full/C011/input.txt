package com.acme.insurance.claims.dto;

import com.fasterxml.jackson.annotation.JsonInclude;
import com.fasterxml.jackson.annotation.JsonProperty;
import lombok.Data;
import lombok.NoArgsConstructor;
import lombok.AllArgsConstructor;
import java.util.List;

/**
 * Data Transfer Object representing the system's outbound response for a claim.
 * Strips away internal data types in favor of consumer-friendly terminology.
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
@JsonInclude(JsonInclude.Include.NON_NULL)
public class ClaimResponse {

    @JsonProperty("claim_id")
    private String claimId;

    @JsonProperty("claim_status")
    private String claimStatus;

    @JsonProperty("attached_document_ids")
    private List<String> attachedDocumentIds;

    @JsonProperty("error_message")
    private String errorMessage;

    /**
     * Evaluates if the response represents an active, ongoing claim.
     */
    public boolean isActive() {
        if (claimStatus == null) {
            return false;
        }
        return !claimStatus.equals("SYSTEM_ERROR") && !claimStatus.equals("INVALID_PAYLOAD") && !claimStatus.equals("REJECTED");
    }
}
