package com.acme.insurance.claims.dto;

import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.Size;
import com.fasterxml.jackson.annotation.JsonProperty;
import lombok.Data;
import lombok.NoArgsConstructor;
import java.util.List;

/**
 * Payload for updating an existing claim with supplementary documents.
 */
@Data
@NoArgsConstructor
public class DocumentUpdateRequest {

    @NotEmpty(message = "At least one document ID must be provided")
    @Size(max = 10, message = "A maximum of 10 documents can be attached per request")
    @JsonProperty("document_ids")
    private List<String> documentIds;

}
