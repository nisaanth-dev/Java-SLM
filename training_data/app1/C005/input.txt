package com.aegisinsure.underwriting.controller;

import com.aegisinsure.underwriting.dto.request.PolicyApplicationRequest;
import com.aegisinsure.underwriting.dto.response.PolicyApplicationResponse;
import com.aegisinsure.underwriting.dto.response.QuoteResponse;
import com.aegisinsure.underwriting.entity.ApplicationStatus;
import com.aegisinsure.underwriting.service.PolicyApplicationService;
import com.aegisinsure.underwriting.service.QuoteService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.util.UriComponentsBuilder;

import java.net.URI;
import java.util.List;

/**
 * REST API for auto policy applications and the quotes derived from them.
 * Applications are the entry point of the underwriting funnel; quotes are
 * modelled as a sub-resource of the application that produced them.
 */
@RestController
@RequestMapping("/api/v1/applications")
@RequiredArgsConstructor
public class PolicyApplicationController {

    private final PolicyApplicationService applicationService;
    private final QuoteService quoteService;

    /**
     * Creates and submits a new policy application. Returns 201 with a Location
     * header pointing at the created resource.
     */
    @PostMapping
    public ResponseEntity<PolicyApplicationResponse> create(
            @Valid @RequestBody PolicyApplicationRequest request,
            UriComponentsBuilder uriBuilder) {

        PolicyApplicationResponse created = applicationService.createApplication(request);
        URI location = uriBuilder.path("/api/v1/applications/{reference}")
                .buildAndExpand(created.applicationReference())
                .toUri();
        return ResponseEntity.created(location).body(created);
    }

    /**
     * Retrieves a single application by its business reference.
     */
    @GetMapping("/{reference}")
    public ResponseEntity<PolicyApplicationResponse> getByReference(@PathVariable String reference) {
        return ResponseEntity.ok(applicationService.getByReference(reference));
    }

    /**
     * Lists applications filtered by lifecycle status.
     */
    @GetMapping
    public ResponseEntity<List<PolicyApplicationResponse>> listByStatus(
            @RequestParam ApplicationStatus status) {
        return ResponseEntity.ok(applicationService.getByStatus(status));
    }

    /**
     * Withdraws an application at the applicant's request.
     */
    @PostMapping("/{reference}/withdrawal")
    public ResponseEntity<PolicyApplicationResponse> withdraw(@PathVariable String reference) {
        return ResponseEntity.ok(applicationService.withdraw(reference));
    }

    /**
     * Generates a quote for an approved application. Returns 201 for the newly
     * issued quote.
     */
    @PostMapping("/{reference}/quotes")
    public ResponseEntity<QuoteResponse> generateQuote(@PathVariable String reference,
                                                       UriComponentsBuilder uriBuilder) {
        QuoteResponse quote = quoteService.generateQuote(reference);
        URI location = uriBuilder.path("/api/v1/applications/{reference}/quotes/{quoteReference}")
                .buildAndExpand(reference, quote.quoteReference())
                .toUri();
        return ResponseEntity.created(location).body(quote);
    }

    /**
     * Lists every quote generated for an application, newest first.
     */
    @GetMapping("/{reference}/quotes")
    public ResponseEntity<List<QuoteResponse>> listQuotes(@PathVariable String reference) {
        return ResponseEntity.ok(quoteService.getByApplicationReference(reference));
    }

    /**
     * Accepts an issued quote so it can be bound into a policy.
     */
    @PostMapping("/{reference}/quotes/{quoteReference}/acceptance")
    public ResponseEntity<QuoteResponse> acceptQuote(@PathVariable String reference,
                                                     @PathVariable String quoteReference) {
        return ResponseEntity.ok(quoteService.acceptQuote(quoteReference));
    }

    /**
     * Rejects an issued quote.
     */
    @PostMapping("/{reference}/quotes/{quoteReference}/rejection")
    public ResponseEntity<QuoteResponse> rejectQuote(@PathVariable String reference,
                                                     @PathVariable String quoteReference) {
        return ResponseEntity.ok(quoteService.rejectQuote(quoteReference));
    }
}
