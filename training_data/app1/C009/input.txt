package com.aegisinsure.underwriting.dto.request;

import com.aegisinsure.underwriting.entity.CoverageType;
import jakarta.validation.constraints.DecimalMin;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.PositiveOrZero;

import java.math.BigDecimal;

/**
 * Inbound payload describing a single requested coverage line: the coverage
 * type, the desired limit, and the chosen deductible.
 */
public record CoverageSelectionRequest(

        @NotNull(message = "Coverage type is required")
        CoverageType coverageType,

        @NotNull(message = "Coverage limit is required")
        @DecimalMin(value = "1000.00", message = "Coverage limit must be at least 1000.00")
        BigDecimal coverageLimit,

        @PositiveOrZero(message = "Deductible must not be negative")
        BigDecimal deductible
) {
}
