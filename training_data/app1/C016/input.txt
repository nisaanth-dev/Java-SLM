/**
 * Read model returned for a premium quote. Exposes the rated premium, the
 * fee/tax breakdown, the installment plan, and a per-coverage premium
 * breakdown via the nested {@link PremiumLineView}.
 */
public record QuoteResponse(
        Long id,
        String quoteReference,
        String applicationReference,
        QuoteStatus status,
        BigDecimal annualPremium,
        BigDecimal fees,
        BigDecimal taxes,
        BigDecimal totalPayable,
        int installmentCount,
        BigDecimal installmentAmount,
        Instant issuedAt,
        Instant expiresAt,
        List<PremiumLineView> premiumLines
) {

    /**
     * Nested per-coverage premium breakdown line.
     */
    public record PremiumLineView(
            String coverageType,
            BigDecimal calculatedPremium
    ) {
    }
}
