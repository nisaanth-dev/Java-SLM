package com.aegisinsure.underwriting.event.publisher;

import com.aegisinsure.underwriting.config.KafkaConfig;
import com.aegisinsure.underwriting.entity.PolicyApplication;
import com.aegisinsure.underwriting.event.DomainEvent;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.stereotype.Component;

import java.util.HashMap;
import java.util.Map;

/**
 * Publishes application lifecycle domain events to the underwriting application
 * lifecycle topic. Used by the underwriting service to broadcast review,
 * approval, referral, and decline outcomes to interested downstream consumers.
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class UnderwritingEventPublisher {

    private final KafkaTemplate<String, Object> kafkaTemplate;

    /**
     * Publishes an underwriting decision event for the supplied application.
     * The application reference is used as the partition key so all events for
     * one application preserve ordering.
     */
    public void publishDecision(PolicyApplication application, String eventType) {
        if (application == null || application.getApplicationReference() == null) {
            log.warn("Skipping publish: application or reference is null");
            return;
        }

        Map<String, Object> payload = new HashMap<>();
        payload.put("applicationReference", application.getApplicationReference());
        payload.put("status", application.getStatus() == null ? null : application.getStatus().name());
        payload.put("riskScore", application.getRiskScore());
        payload.put("riskTier", application.getRiskTier() == null ? null : application.getRiskTier().name());
        payload.put("decisionReason", application.getDecisionReason());

        DomainEvent event = DomainEvent.of(
                eventType, "PolicyApplication", application.getApplicationReference(), payload);

        dispatch(application.getApplicationReference(), event);
    }

    /**
     * Publishes a lightweight status-change event without decision metadata.
     */
    public void publishStatusChange(String applicationReference, String newStatus) {
        if (applicationReference == null) {
            log.warn("Skipping status-change publish: reference is null");
            return;
        }
        Map<String, Object> payload = new HashMap<>();
        payload.put("applicationReference", applicationReference);
        payload.put("status", newStatus);

        DomainEvent event = DomainEvent.of(
                "APPLICATION_STATUS_CHANGED", "PolicyApplication", applicationReference, payload);
        dispatch(applicationReference, event);
    }

    /**
     * Sends the event to Kafka, logging success and translating send failures
     * into a logged warning rather than propagating to the caller.
     */
    private void dispatch(String key, DomainEvent event) {
        try {
            kafkaTemplate.send(KafkaConfig.APPLICATION_LIFECYCLE_TOPIC, key, event);
            log.debug("Published {} for application {}", event.eventType(), key);
        } catch (RuntimeException ex) {
            log.error("Failed to publish {} for application {}: {}",
                    event.eventType(), key, ex.getMessage());
        }
    }
}
