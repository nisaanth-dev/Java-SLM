package com.aegisinsure.underwriting.filter;

import jakarta.servlet.Filter;
import jakarta.servlet.FilterChain;
import jakarta.servlet.ServletException;
import jakarta.servlet.ServletRequest;
import jakarta.servlet.ServletResponse;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.slf4j.MDC;
import org.springframework.core.Ordered;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

import java.io.IOException;
import java.util.UUID;

/**
 * Servlet filter that establishes a correlation id for every inbound HTTP
 * request. An existing {@code X-Correlation-Id} header is honoured; otherwise a
 * fresh identifier is generated. The id is placed in the SLF4J {@link MDC} so it
 * is stamped on every log line for the request, echoed back on the response, and
 * always cleared in a finally block to prevent leakage across pooled threads.
 */
@Component
@Order(Ordered.HIGHEST_PRECEDENCE)
public class CorrelationIdFilter implements Filter {

    /** Header carrying the cross-service request correlation id. */
    public static final String CORRELATION_ID_HEADER = "X-Correlation-Id";

    /** MDC key under which the correlation id is exposed to the logging layer. */
    public static final String CORRELATION_ID_MDC_KEY = "correlationId";

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {

        // Resolve an inbound correlation id or mint a new one for this request.
        String correlationId = resolveCorrelationId(request);
        MDC.put(CORRELATION_ID_MDC_KEY, correlationId);

        // Echo the id on the response so callers can stitch their own traces.
        if (response instanceof HttpServletResponse httpResponse) {
            httpResponse.setHeader(CORRELATION_ID_HEADER, correlationId);
        }

        try {
            chain.doFilter(request, response);
        } finally {
            // Always clear the MDC so the value never leaks to the next request
            // served by this (pooled) worker thread.
            MDC.remove(CORRELATION_ID_MDC_KEY);
        }
    }

    /**
     * Extracts a non-blank correlation id from the request header, falling back
     * to a freshly-generated UUID when none is supplied.
     */
    private String resolveCorrelationId(ServletRequest request) {
        if (request instanceof HttpServletRequest httpRequest) {
            String header = httpRequest.getHeader(CORRELATION_ID_HEADER);
            if (header != null && !header.isBlank()) {
                return header;
            }
        }
        return UUID.randomUUID().toString();
    }
}
